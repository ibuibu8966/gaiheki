'use client';

import { useState, useEffect } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '@/components/ui/table';

          <CardHeader>
            <CardTitle className="text-sm">全加盟店総売上</CardTitle>
          </CardHeader>
          <CardContent>
            <p className="text-3xl font-bold">
              ¥{dashboardData.kpi.total_revenue.toLocaleString()}
            </p>
          </CardContent>
        </Card>
        <Card>
          <CardHeader>
            <CardTitle className="text-sm">手数料売上</CardTitle>
          </CardHeader>

        </Card>

            <CardTitle className="text-sm">新規加盟店数</CardTitle>

        <CardHeader>
          <CardTitle>加盟店別サマリー</CardTitle>
        </CardHeader>
        <CardContent>
          <Table>
            <TableHeader>
              <TableRow>

### 2. 手数料請求管理画面

#### 2.1 ページ作成

`app/admin/billing/page.tsx` を作成してください。

#### 2.2 実装内容

**請求書一覧テーブル**
- 請求書番号、加盟店名、請求期間、請求額、発行日、支払期日、ステータスを表示
- ステータスでフィルタリング
- 加盟店でフィルタリング

**アクションボタン**
- 「請求データ生成」ボタン（右上）
- 「一括発行」ボタン（下書き状態の請求書を一括発行）
- 各行に「詳細」「PDF出力」ボタン

**請求データ生成モーダル**
- 請求期間（開始日・終了日）を入力
- 「生成」ボタンで請求データを自動生成
- 生成結果をプレビュー表示

#### 2.3 実装例

```tsx
'use client';

import { useState, useEffect } from 'react';
import { Button } from '@/components/ui/button';
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '@/components/ui/table';
import { Badge } from '@/components/ui/badge';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogTrigger } from '@/components/ui/dialog';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';

export default function BillingManagement() {
  const [invoices, setInvoices] = useState([]);
  const [status, setStatus] = useState('all');
  const [isGenerateModalOpen, setIsGenerateModalOpen] = useState(false);
  const [billingPeriod, setBillingPeriod] = useState({
    start: '',
    end: '',
  });
  const [generatedInvoices, setGeneratedInvoices] = useState(null);

  useEffect(() => {
    fetchInvoices();
  }, [status]);

  const fetchInvoices = async () => {
    const params = new URLSearchParams();
    if (status !== 'all') params.append('status', status);
    
    const res = await fetch(`/api/admin/invoices?${params}`);
    const data = await res.json();
    setInvoices(data.invoices);
  };

  const handleGenerateInvoices = async () => {
    const res = await fetch('/api/admin/invoices/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        billing_period_start: billingPeriod.start,
        billing_period_end: billingPeriod.end,
      }),
    });
    const data = await res.json();
    setGeneratedInvoices(data);
    fetchInvoices();
  };

  const handleIssueInvoices = async () => {
    const draftInvoices = invoices.filter((inv) => inv.status === 'DRAFT');
    const invoiceIds = draftInvoices.map((inv) => inv.id);
    
    await fetch('/api/admin/invoices/issue', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ invoice_ids: invoiceIds }),
    });
    
    fetchInvoices();
  };

  return (
    <div className="p-6 space-y-6">
      <div className="flex justify-between items-center">
        <h1 className="text-3xl font-bold">手数料請求管理</h1>
        <div className="flex gap-2">
          <Dialog open={isGenerateModalOpen} onOpenChange={setIsGenerateModalOpen}>
            <DialogTrigger asChild>
              <Button>請求データ生成</Button>
            </DialogTrigger>
            <DialogContent>
              <DialogHeader>
                <DialogTitle>請求データ生成</DialogTitle>
              </DialogHeader>
              <div className="space-y-4">
                <div>
                  <Label>請求期間（開始日）</Label>
                  <Input
                    type="date"
                    value={billingPeriod.start}
                    onChange={(e) => setBillingPeriod({ ...billingPeriod, start: e.target.value })}
                  />
                </div>
                <div>
                  <Label>請求期間（終了日）</Label>
                  <Input
                    type="date"
                    value={billingPeriod.end}
                    onChange={(e) => setBillingPeriod({ ...billingPeriod, end: e.target.value })}
                  />
                </div>
                <Button onClick={handleGenerateInvoices} className="w-full">
                  生成
                </Button>
                
                {generatedInvoices && (
                  <div className="mt-4">
                    <p className="font-bold">{generatedInvoices.generated}件の請求書を生成しました</p>
                  </div>
                )}
              </div>
            </DialogContent>
          </Dialog>
          
          <Button onClick={handleIssueInvoices} variant="outline">
            一括発行
          </Button>
        </div>
      </div>

      <Table>
        <TableHeader>
          <TableRow>
            <TableHead>請求書番号</TableHead>
            <TableHead>加盟店名</TableHead>
            <TableHead>請求期間</TableHead>
            <TableHead className="text-right">請求額</TableHead>
            <TableHead>発行日</TableHead>
            <TableHead>支払期日</TableHead>
            <TableHead>ステータス</TableHead>
            <TableHead>操作</TableHead>
          </TableRow>
        </TableHeader>
        <TableBody>
          {invoices.map((invoice) => (
            <TableRow key={invoice.id}>
              <TableCell>{invoice.invoice_number}</TableCell>
              <TableCell>{invoice.company_name}</TableCell>
              <TableCell>
                {invoice.billing_period_start} 〜 {invoice.billing_period_end}
              </TableCell>
              <TableCell className="text-right">
                ¥{invoice.grand_total.toLocaleString()}
              </TableCell>
              <TableCell>{invoice.issue_date}</TableCell>
              <TableCell>{invoice.due_date}</TableCell>
              <TableCell>
                <Badge>{invoice.status}</Badge>
              </TableCell>
              <TableCell>
                <Button variant="outline" size="sm">詳細</Button>
              </TableCell>
            </TableRow>
          ))}
        </TableBody>
      </Table>
    </div>
  );
}
```

### 3. 設定画面

#### 3.1 ページ作成

`app/admin/settings/page.tsx` を作成してください。

#### 3.2 実装内容

**タブ切り替え**
- 「料金プラン設定」タブ

**料金プラン設定タブ**
- 料金プラン一覧を表示

- 各プランの編集・削除ボタン
- デフォルトプランの設定

**システム設定タブ**
- 請求締め日の設定
- 支払期日の設定
- 消費税率の設定

#### 3.3 実装例

```tsx
'use client';

import { useState, useEffect } from 'react';
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '@/components/ui/table';

export default function Settings() {
  const [feePlans, setFeePlans] = useState([]);
  const [systemSettings, setSystemSettings] = useState(null);

  useEffect(() => {
    fetchFeePlans();
    fetchSystemSettings();
  }, []);

  const fetchFeePlans = async () => {
    const res = await fetch('/api/admin/settings/fee-plans');
    const data = await res.json();
    setFeePlans(data.plans);
  };

  const fetchSystemSettings = async () => {
    const res = await fetch('/api/admin/settings/system');
    const data = await res.json();
    setSystemSettings(data);
  };

  const handleUpdateSystemSettings = async () => {
    await fetch('/api/admin/settings/system', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(systemSettings),
    });
    alert('設定を更新しました');
  };

  return (
    <div className="p-6 space-y-6">
      <h1 className="text-3xl font-bold">設定</h1>

      <Tabs defaultValue="fee-plans">
        <TabsList>
          <TabsTrigger value="fee-plans">料金プラン設定</TabsTrigger>
          <TabsTrigger value="system">システム設定</TabsTrigger>
        </TabsList>

        <TabsContent value="fee-plans">
          <Card>
            <CardHeader>
              <div className="flex justify-between items-center">
                <CardTitle>料金プラン一覧</CardTitle>
                <Button>新規作成</Button>
              </div>
            </CardHeader>
            <CardContent>
              <Table>
                <TableHeader>
                  <TableRow>
                    <TableHead>プラン名</TableHead>
                    <TableHead className="text-right">月額固定</TableHead>
                    <TableHead className="text-right">受注ごと</TableHead>
                    <TableHead className="text-right">施工完了ごと</TableHead>
                    <TableHead className="text-right">施工完了料率</TableHead>
                    <TableHead>デフォルト</TableHead>
                    <TableHead>操作</TableHead>
                  </TableRow>
                </TableHeader>
                <TableBody>
                  {feePlans.map((plan) => (
                    <TableRow key={plan.id}>
                      <TableCell>{plan.name}</TableCell>
                      <TableCell className="text-right">
                        {plan.monthly_fee ? `¥${plan.monthly_fee.toLocaleString()}` : '-'}
                      </TableCell>
                      <TableCell className="text-right">
                        {plan.per_order_fee ? `¥${plan.per_order_fee.toLocaleString()}` : '-'}
                      </TableCell>
                      <TableCell className="text-right">
                        {plan.per_project_fee ? `¥${plan.per_project_fee.toLocaleString()}` : '-'}
                      </TableCell>
                      <TableCell className="text-right">
                        {plan.project_fee_rate ? `${(plan.project_fee_rate * 100).toFixed(1)}%` : '-'}
                      </TableCell>
                      <TableCell>{plan.is_default ? '✓' : ''}</TableCell>
                      <TableCell>
                        <Button variant="outline" size="sm">編集</Button>
                      </TableCell>
                    </TableRow>
                  ))}
                </TableBody>
              </Table>
            </CardContent>
          </Card>
        </TabsContent>

        <TabsContent value="system">
          {systemSettings && (
            <Card>
              <CardHeader>
                <CardTitle>システム設定</CardTitle>
              </CardHeader>
              <CardContent className="space-y-4">
                <div>
                  <Label>請求締め日</Label>
                  <Input
                    type="number"
                    value={systemSettings.billing_cycle_closing_day}
                    onChange={(e) => setSystemSettings({
                      ...systemSettings,
                      billing_cycle_closing_day: parseInt(e.target.value)
                    })}
                  />
                  <p className="text-sm text-gray-500 mt-1">月末の場合は31を入力</p>
                </div>
                <div>
                  <Label>支払期日（翌月の日付）</Label>
                  <Input
                    type="number"
                    value={systemSettings.billing_cycle_payment_day}
                    onChange={(e) => setSystemSettings({
                      ...systemSettings,
                      billing_cycle_payment_day: parseInt(e.target.value)
                    })}
                  />
                  <p className="text-sm text-gray-500 mt-1">例: 20日払いの場合は20を入力</p>
                </div>
                <div>
                  <Label>消費税率</Label>
                  <Input
                    type="number"
                    step="0.01"
                    value={systemSettings.tax_rate}
                    onChange={(e) => setSystemSettings({
                      ...systemSettings,
                      tax_rate: parseFloat(e.target.value)
                    })}
                  />
                  <p className="text-sm text-gray-500 mt-1">10%の場合は0.10を入力</p>
                </div>
                <Button onClick={handleUpdateSystemSettings}>設定を保存</Button>
              </CardContent>
            </Card>
          )}
        </TabsContent>
      </Tabs>
    </div>
  );
}
```
